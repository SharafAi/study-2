//core modulees
const fs = require('fs');
const path = require('path');
const rootDir = require('../utilities/pathUtil');


module.exports = class Home { 
  constructor(houseName, price, location, rating, photoURL){
    this.houseName = houseName;
    this.price = price;
    this.location = location;
    this.rating = rating;
    this.photoURL = photoURL;
  }
  
  save() {
    this.id = Math.random().toString();
    Home.fetchAll((RergisterdHomes) => {
      RergisterdHomes.push(this);
      const homeDataPath = path.join(rootDir, 'data', 'homes.json');
      fs.writeFile(homeDataPath, JSON.stringify(RergisterdHomes), (error) => {
        console.log("file writing concluded", error);
      })
    });
  }

  static fetchAll(callback) {
    const homeDataPath = path.join(rootDir, 'data', 'homes.json');
    fs.readFile(homeDataPath, (err, data) => {
      callback(!err ? JSON.parse(data) : []);
    });
  }
}

// readProject.js
// Run with: node readProject.js

const fs = require("fs");
const path = require("path");

const projectRoot = __dirname; 
const outputFile = path.join(projectRoot, "project-dump.txt");

// folders/files to ignore
const ignoreList = ["node_modules", ".git", "project-dump.txt"];

// Function to recursively read all files
function readFolder(folderPath, output = []) {
  const items = fs.readdirSync(folderPath);

  items.forEach((item) => {
    if (ignoreList.includes(item)) return; // skip ignored items

    const fullPath = path.join(folderPath, item);
    const stats = fs.statSync(fullPath);

    if (stats.isDirectory()) {
      readFolder(fullPath, output); // go deeper
    } else {
      try {
        const content = fs.readFileSync(fullPath, "utf-8");
        output.push(
          `==== File: ${fullPath} ====\n\n${content}\n\n`
        );
      } catch (err) {
        output.push(
          `==== File: ${fullPath} (Unreadable) ====\n\n[Error reading file]\n\n`
        );
      }
    }
  });

  return output;
}

// Main execution
const allFilesContent = readFolder(projectRoot, []).join("\n");
fs.writeFileSync(outputFile, allFilesContent, "utf-8");

console.log(`✅ Project dumped to ${outputFile}`);
